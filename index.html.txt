  <h2>Auto Refresh Website</h2>
  <input type="text" id="url" placeholder="Enter website URL">
  <input type="number" id="time" placeholder="Refresh time (seconds)">
  <button onclick="startRefresh()">Start</button>
  <button onclick="stopRefresh()">Stop</button>
  <iframe id="frame"></iframe>
</div>

<script>
  let interval;

  function startRefresh() {
    const url = document.getElementById('url').value;
    const time = document.getElementById('time').value * 1000;

    if (!url || !time) {
      alert('Enter URL and time');
      return;
    }

    const frame = document.getElementById('frame');
    frame.src = url;

    interval = setInterval(() => {
      frame.src = frame.src;
    }, time);
  }

  function stopRefresh() {
    clearInterval(interval);
  }
</script>
